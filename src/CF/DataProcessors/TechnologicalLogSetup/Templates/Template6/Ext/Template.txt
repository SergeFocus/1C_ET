Name:
Administrator actions and errors
Details:
The template describes a small-sized technological log that stores the following operations: starting and stopping applications, connecting to and disconnecting from 1C:Enterprise servers, as well as exceptions generated by 1C:Enterprise. This technological log is usually sufficient for investigating exceptions raised both in the configuration and in 1C:Enterprise technological platform.
Actions:
PathToTL
Template:
<?xml version="1.0" encoding="UTF-8"?>
<config xmlns="http://v8.1c.ru/v8/tech-log">
	<dump create="false"/>
	<log location="c:\v82\logs" history="168">
		<event>
			<eq property="name" value="proc"/>
		</event>
		<event>
			<eq property="name" value="scom"/>
		</event>
		<event>
			<eq property="name" value="conn"/>
		</event>
		<event>
			<eq property="name" value="admin"/>
		</event>
		<event>
			<eq property="name" value="excp"/>
		</event>
		<event>
			<eq property="name" value="qerr"/>
		</event>
		<property name="all"/>
	</log>
</config>
